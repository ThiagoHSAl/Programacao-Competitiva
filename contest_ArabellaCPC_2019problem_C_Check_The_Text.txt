#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <ctype.h>
 
int main()
{
  int n,teclas,i,posicao=0,tamanho,caps=-1;
  char entrada[2001],check[2001],linha[10];
  scanf("%d",&n);
  getchar();
  fgets(entrada,sizeof(entrada),stdin);
  tamanho=strlen(entrada);
  entrada[tamanho-1]='\0';
  scanf("%d",&teclas);
  for(i=0;i<teclas;i++)
  {
    scanf("%s",linha);
    if(strcmp(linha,"CapsLock")==0)
    {
      caps*=-1;
    }
 
    else if(strcmp(linha,"Backspace")==0)
    {
        if(posicao>=1)
        {
          posicao--;
          check[posicao]='@';
        }
    }
    else if(strcmp(linha,"Space")==0)
    {
        check[posicao]=' ';
        posicao++;
    }
    else
    {
        if(caps==1)
            linha[0]=toupper(linha[0]);
     check[posicao]=linha[0];
     posicao++;
    }
  }
  tamanho=strlen(entrada);
  check[tamanho]='\0';
  if(strcmp(check,entrada)==0)
    printf("Correct");
  else
    printf("Incorrect");
}
