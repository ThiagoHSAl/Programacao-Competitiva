#include <iostream>
#include<list>
#include<string>
#include<algorithm>
#include<vector>
#include<set>
using namespace std;
 
int main()
{
    int n,i;
    string nome;
    vector<string> chat(200000);
    set<string> print;
    cin>>n;
    for(i=0;i<n;i++)
    {
        cin>>nome;
        chat[i]=nome;
    }
    for(i=n-1;i>=0;i--)
    {
       if(print.find(chat[i])==print.end())
       {
           print.insert(chat[i]);
           cout<<chat[i]<<endl;
       }
    }
    return 0;
}
